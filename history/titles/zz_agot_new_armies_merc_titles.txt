d_laamp_merc_flame_archers = {
	8082.1.1 = {
		liege = 0
		holder = AGNA_1
		government = landless_adventurer_government
		succession_laws = { landless_adventurer_succession_law }
		effect = {
			holder ?= {
				if = {
					limit = {
						NOT = { has_realm_law = camp_purpose_mercenaries }
					}
					add_realm_law_skip_effects = camp_purpose_mercenaries
				}
				create_maa_regiment = { type = agot_flame_archers_essos check_can_recruit = no size = 9 }
			}
			create_landless_adventurer_title_history_effect = yes
			set_variable = { name = adventurer_creation_reason value = flag:historical }
			if = {
				limit = { NOT = { has_dlc_feature = roads_to_power } }
				holder ?= { destroy_title = prev }
			}
			else = { holder ?= { add_character_flag = block_laamp_starting_event } }
		}
	}
}
