namespace = agot_new_armies

# Hidden monthly logic:
# - Save sync: apply mercenary unlock traditions in ongoing saves
# - Direct mercenary company spawn/repair for ongoing saves
# - Flame Archers: random unstable fire/backfire effects
# - Second Sons: penalties when ruler is poor
agot_new_armies.0000 = {
	type = character_event
	hidden = yes

	trigger = {
		is_player = yes
		is_alive = yes
	}

	immediate = {
		culture:pentoshi = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_flame_archers } }
				add_culture_tradition = tradition_agot_new_armies_flame_archers
			}
		}
		culture:myrish = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_flame_archers } }
				add_culture_tradition = tradition_agot_new_armies_flame_archers
			}
		}
		culture:moon_clan = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_mountain_clans } }
				add_culture_tradition = tradition_agot_new_armies_mountain_clans
			}
		}
		culture:wolfswood_clansman = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_mountain_clans } }
				add_culture_tradition = tradition_agot_new_armies_mountain_clans
			}
		}
		culture:ironborn = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_assault_divers } }
				add_culture_tradition = tradition_agot_new_armies_assault_divers
			}
		}
		culture:braavosi = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_second_sons } }
				add_culture_tradition = tradition_agot_new_armies_second_sons
			}
		}
		culture:northman = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_monolith_warriors } }
				add_culture_tradition = tradition_agot_new_armies_monolith_warriors
			}
		}
		culture:riverman_main = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_brotherhood_without_banners } }
				add_culture_tradition = tradition_agot_new_armies_brotherhood_without_banners
			}
		}
		culture:crownlander = {
			if = {
				limit = { NOT = { has_culture_tradition = tradition_agot_new_armies_brotherhood_without_banners } }
				add_culture_tradition = tradition_agot_new_armies_brotherhood_without_banners
			}
		}
	}
}

agot_new_armies.0001 = {
	type = character_event
	hidden = yes

	trigger = {
		is_ruler = yes
		is_alive = yes
	}

	immediate = {
		if = {
			limit = {
				any_maa_regiment = { is_maa_type = agot_flame_archers_essos }
			}
			random_list = {
				10 = {
					add_character_modifier = {
						modifier = agot_flame_archers_rhllor_zeal
						days = 30
					}
				}
				10 = {
					add_character_modifier = {
						modifier = agot_flame_archers_unstable_fire
						days = 30
					}
				}
				80 = { }
			}
		}

		if = {
			limit = {
				any_maa_regiment = { is_maa_type = agot_second_sons }
				NOT = { gold >= 100 }
			}
			add_character_modifier = {
				modifier = agot_second_sons_unpaid
				days = 30
			}
		}

		if = {
			limit = {
				any_maa_regiment = { is_maa_type = agot_monolith_warriors }
			}
			random_list = {
				15 = {
					add_character_modifier = {
						modifier = agot_monolith_anchor
						days = 30
					}
				}
				8 = {
					add_character_modifier = {
						modifier = agot_monolith_hammer_impact
						days = 30
					}
				}
				77 = { }
			}
		}

		if = {
			limit = {
				any_maa_regiment = { is_maa_type = agot_brotherhood_without_banners }
				is_at_war = yes
			}
			random_list = {
				20 = {
					add_character_modifier = {
						modifier = agot_bwb_flaming_swords
						days = 30
					}
				}
				10 = {
					add_character_modifier = {
						modifier = agot_bwb_brought_back
						days = 30
					}
				}
				70 = { }
			}
		}

		if = {
			limit = {
				any_maa_regiment = { is_maa_type = agot_brotherhood_without_banners }
				is_at_war = no
			}
			random_list = {
				20 = {
					add_character_modifier = {
						modifier = agot_bwb_unruly_company
						days = 30
					}
				}
				80 = { }
			}
		}
	}
}

agot_new_armies.0101 = {
	type = character_event
	title = agot_new_armies.0101.t
	desc = agot_new_armies.0101.desc
	is_triggered_only = yes

	trigger = {
		is_player = yes
		is_landed = yes
		is_alive = yes
		NOT = { has_character_flag = agot_new_armies_contracts_dismissed }
	}

	option = {
		name = agot_new_armies.0101.a
		trigger_event = { id = agot_new_armies.0100 }
	}

	option = {
		name = agot_new_armies.0101.b
		add_character_flag = agot_new_armies_contracts_dismissed
	}
}

# Hidden spawner – creates the 6 mercenary companies as hireable LAAMP entities.
# Called from the decision AND monthly pulse so companies stay alive.
agot_new_armies.0300 = {
	type = character_event
	hidden = yes
	is_triggered_only = yes

	immediate = {
		# -- Arqueiros de Chamas de Essos --
		if = {
			limit = { NOT = { title:d_laamp_merc_flame_archers = { exists = holder } } }
			create_character = {
				culture = culture:myrish
				religion = religion:rhllor_fots
				trait = education_martial_3
				trait = ambitious
				trait = immortal
				age = 35
				save_scope_as = agot_cap1_scope
			}
			title:d_laamp_merc_flame_archers = {
				change_title_holder = { holder = scope:agot_cap1_scope }
			}
			scope:agot_cap1_scope = {
				change_government = landless_adventurer_government
				add_realm_law_skip_effects = camp_purpose_mercenaries
				create_maa_regiment = { type = agot_flame_archers_essos check_can_recruit = no size = 9 }
			}
		}

		# -- Clãs das Montanhas --
		if = {
			limit = { NOT = { title:d_laamp_merc_mountain_clans = { exists = holder } } }
			create_character = {
				culture = culture:wolfswood_clansman
				religion = religion:fots_seven
				trait = education_martial_2
				trait = brave
				trait = immortal
				age = 35
				save_scope_as = agot_cap2_scope
			}
			title:d_laamp_merc_mountain_clans = {
				change_title_holder = { holder = scope:agot_cap2_scope }
			}
			scope:agot_cap2_scope = {
				change_government = landless_adventurer_government
				add_realm_law_skip_effects = camp_purpose_mercenaries
				create_maa_regiment = { type = agot_mountain_clans check_can_recruit = no size = 12 }
			}
		}

		# -- Mergulhadores de Assalto --
		if = {
			limit = { NOT = { title:d_laamp_merc_assault_divers = { exists = holder } } }
			create_character = {
				culture = culture:ironborn
				religion = religion:fots_drowned
				trait = education_martial_3
				trait = brave
				trait = immortal
				age = 35
				save_scope_as = agot_cap3_scope
			}
			title:d_laamp_merc_assault_divers = {
				change_title_holder = { holder = scope:agot_cap3_scope }
			}
			scope:agot_cap3_scope = {
				change_government = landless_adventurer_government
				add_realm_law_skip_effects = camp_purpose_mercenaries
				create_maa_regiment = { type = agot_ironborn_assault_divers check_can_recruit = no size = 25 }
			}
		}

		# -- Segundos Filhos --
		if = {
			limit = { NOT = { title:d_laamp_merc_second_sons_new = { exists = holder } } }
			create_character = {
				culture = culture:myrish
				religion = religion:fc_pan_myr
				trait = education_martial_3
				trait = calm
				trait = immortal
				age = 35
				save_scope_as = agot_cap4_scope
			}
			title:d_laamp_merc_second_sons_new = {
				change_title_holder = { holder = scope:agot_cap4_scope }
			}
			scope:agot_cap4_scope = {
				change_government = landless_adventurer_government
				add_realm_law_skip_effects = camp_purpose_mercenaries
				create_maa_regiment = { type = agot_second_sons check_can_recruit = no size = 60 }
			}
		}

		# -- Guerreiros Monólito --
		if = {
			limit = { NOT = { title:d_laamp_merc_monolith_guard = { exists = holder } } }
			create_character = {
				culture = culture:northman
				religion = religion:fots_seven
				trait = education_martial_4
				trait = stubborn
				trait = immortal
				age = 35
				save_scope_as = agot_cap5_scope
			}
			title:d_laamp_merc_monolith_guard = {
				change_title_holder = { holder = scope:agot_cap5_scope }
			}
			scope:agot_cap5_scope = {
				change_government = landless_adventurer_government
				add_realm_law_skip_effects = camp_purpose_mercenaries
				create_maa_regiment = { type = agot_monolith_warriors check_can_recruit = no size = 10 }
			}
		}

		# -- Irmandade Sem Bandeiras --
		if = {
			limit = { NOT = { title:d_laamp_merc_brotherhood_without_banners = { exists = holder } } }
			create_character = {
				culture = culture:riverman_main
				religion = religion:rhllor_fots
				trait = education_martial_3
				trait = brave
				trait = immortal
				age = 35
				save_scope_as = agot_cap6_scope
			}
			title:d_laamp_merc_brotherhood_without_banners = {
				change_title_holder = { holder = scope:agot_cap6_scope }
			}
			scope:agot_cap6_scope = {
				change_government = landless_adventurer_government
				add_realm_law_skip_effects = camp_purpose_mercenaries
				create_maa_regiment = { type = agot_brotherhood_without_banners check_can_recruit = no size = 5 }
			}
		}
	}
